import { Playground, PropsTable } from 'docz'
import Percent from './index.js'

# 金额输入框

<PropsTable of={Percent} />

## 基本展示

默认展示2位小数，通过四舍五入取值

<Playground>
    <Percent value={1000.11}/>
    <br/>
    <Percent value={2000.99}/>
</Playground>

## 保留小数点值

通过设置 decimal 属性，可以设置展示到小数位数

<Playground>
    <Percent value={1000.115} decimal={1}/>
    <br/>
    <Percent value={2000.9} decimal={3}/>
    <br/>
    <Percent value={3000.9} decimal={0}/>
</Playground>

## 可编辑状态

通过设置 edit 为 true 可以实现点击可编辑状态

<Playground>
    <Percent value={1000.800} edit={true} onValue={value=>alert(value)}/>
    <br/>
    ¥<Percent value={2000.008} edit={true} decimal={2} onValue={value=>alert(value)}/>
</Playground>
